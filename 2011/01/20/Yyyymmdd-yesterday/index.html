昨日のyyyymmdd

Jan 20th, 2011

Option Explicit

On Error Resume Next

Yday=DateAdd(“d”,-1,Date())

‘ 昨日のyyyymmdd

Wscript.Echo Year(Yday) & Right(“0”&Month(Yday),2) &Right(“0”& Day(Yday),2)

LogFolderPath=“D:\Log”

Set oFso = CreateObject(“Scripting.FileSystemObject”)

‘If oFso.FolderExists(LogFolderPath) = False Then

oFso.CreateFolder()

CatchErr “フォルダが作成されました”

‘End If

Set oFso = Nothing

Sub CatchErr(inMesg)

strMesg=Err.Description

If Err.Number = 0 Then

strMesg=inMesg

End If

Wscript.Echo strMesg

End Sub
